#!/bin/bash
set -e

# Navigate to your project directory
cd /home/ubuntu/studynotes || exit

echo "Pulling latest changes..."
git pull

echo "Syncing frontend dist/ to /var/www/accusoft..."
sudo rsync -av --delete dist/ /var/www/studynotes/

echo "Restarting PM2 process..."
cd /home/ubuntu/accusoft/server
git pull
npm i
pm2 restart 0

echo "Deployment finished successfully!"
